### Повторение 1 — Карта структур (уроки 13, 14, 15)

Объяви структуру `Product{Name string; Price int; Qty int}`. Напиши функцию `totalValue(products []Product) int`, которая считает суммарную стоимость склада (сумма `Price * Qty` по всем товарам). Затем напиши `groupByPrice(products []Product) map[int][]string`, группирующую названия товаров по цене. Проверь на нескольких товарах.

> Связка: срезы структур (уроки 13, 15) + карта значений-срезов (урок 14) + перебор (урок 7). Это объединяет все коллекции блока.

```go
package main

import "fmt"

type Product struct {
	Name  string
	Price int
	Qty   int
}

func totalValue(products []Product) int {
	sum := 0
	for _, x := range products {
		sum += x.Price * x.Qty
	}
	return sum
}

func groupByPrice(products []Product) map[int][]string {
	result := map[int][]string{}
	for _, x := range products {
		result[x.Price] = append(result[x.Price], x.Name)
	}
	return result
}

func main() {
	inv := []Product{
		{Name: "Зелье здоровья", Price: 5, Qty: 5},
		{Name: "Зелье маны", Price: 100, Qty: 1},
		{Name: "Щит", Price: 20, Qty: 1},
		{Name: "Меч", Price: 10, Qty: 5},
		{Name: "Топор", Price: 20, Qty: 10},
	}
	fmt.Println(totalValue(inv))
	fmt.Println(groupByPrice(inv))
}
```
Вывод:
```
395
map[5:[Зелье здоровья] 10:[Меч] 20:[Щит Топор] 100:[Зелье маны]]
```
